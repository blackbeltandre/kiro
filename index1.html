    """Detects spooky apparitions in an image."""
    img = cv2.imread(image_path)
    # Run inference
    ghosts = detector.detect(img, confidence=0.75)
    if ghosts:
        print(f"Found {len(ghosts)} apparitions!")
    else:
        print("All clear... for now.")
